    Problem 200: Number of Islands

    Union Find

    def numIslands(self, grid):
    if not grid: return 0
    root = {}
    def find(x):
        root.setdefault(x, x)
        if root[x] != x:
            root[x] = find(root[x])
        return root[x]
    def union(x, y):
        rx, ry = find(x), find(y)
        if rx != ry:
            root[rx] = ry
            return True
        return False

    count = sum(grid[r][c] == '1' for r in range(len(grid))
                for c in range(len(grid[0])))
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == '1':
                for dr, dc in [(1,0),(0,1)]:
                    nr, nc = r+dr, c+dc
                    if 0 <= nr < len(grid) and 0 <= nc < len(grid[0]) and grid[nr][nc] == '1':
                        if union((r,c),(nr,nc)):
                            count -= 1
    return count

    DFS

    def numIslands(self, grid):
    def dfs(r, c):
        if r < 0 or r >= len(grid) or c < 0 or c >= len(grid[0]) or grid[r][c] != '1':
            return
        grid[r][c] = '#'
        for dr, dc in [(1,0),(0,1),(-1,0),(0,-1)]:
            dfs(r+dr, c+dc)

    count = 0
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == '1':
                dfs(r, c)
                count += 1
    return count

    BFS

    from collections import deque
def numIslands(self, grid):
    count = 0
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == '1':
                count += 1
                q = deque([(r,c)])
                grid[r][c] = '#'
                while q:
                    row, col = q.popleft()
                    for dr, dc in [(1,0),(0,1),(-1,0),(0,-1)]:
                        nr, nc = row+dr, col+dc
                        if 0 <= nr < len(grid) and 0 <= nc < len(grid[0]) and grid[nr][nc] == '1':
                            grid[nr][nc] = '#'
                            q.append((nr,nc))
    return count

    Problem 690: Employee Importance

    def getImportance(self, employees, id):
    emp_map = {e.id: e for e in employees}
    def dfs(eid):
        e = emp_map[eid]
        return e.importance + sum(dfs(s) for s in e.subordinates)
    return dfs(id)
