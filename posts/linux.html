    April 10, 2023 语言学习

    Linux

    Linux 常用命令速查。

    文件操作

    ls -la                  # 列出所有文件（含隐藏）
cd /path/to/dir         # 切换目录
pwd                     # 当前目录
mkdir -p dir/subdir     # 递归创建目录
rm -rf dir              # 删除目录（慎用）
cp -r src/ dst/         # 复制目录
mv old new              # 移动/重命名

    文件查看

    cat file.txt
head -n 20 file.txt
tail -f log.txt         # 实时查看日志
grep -r "keyword" .     # 递归搜索
wc -l file.txt          # 统计行数

    进程管理

    ps aux | grep python    # 查找进程
kill -9            # 强制终止
top                     # 实时进程监控
nohup python app.py &   # 后台运行

    权限

    chmod 755 script.sh     # 设置权限
chmod +x script.sh      # 添加执行权限
chown user:group file   # 修改所有者
